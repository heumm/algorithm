##[체육복](https://programmers.co.kr/learn/courses/30/lessons/42862)

```java
import java.util.*;
class Solution {
    public int solution(int n, int[] lost, int[] reserve) {
        int answer = 0;
        Map<Integer, Integer> map = new HashMap<>();    //key:value = 학생 번호:체육복 수 

        for(int i=1; i<=n; i++) {       //모든 학생을 체육복1개씩 가진것으로 치고 삽입
            map.put(i, 1);
        }
        for(int num : reserve) {    //여벌
            map.put(num, 2);
        }

        for(int num2 : lost) {      //분실
            if(map.get(num2) == 2) map.put(num2, 1);     //여벌이 있지만 분실한 학생
            else map.put(num2, 0);
        }
        
        Set<Integer> keySet = map.keySet();
        int max = Collections.max(keySet);
        
        for(int key : keySet) {

            if(key == 1) { //첫 번호의 학생이
                if(map.get(key) == 0) { //체육복이 없는 학생의 경우
                    if(map.get(key+1) == 2) { //다음번호 학생이 여벌이 있으면
                        map.put(key, 1);    //해당 학생에게 체육복 수를 1로맞추고
                        map.put(key+1, 1);  //다음번호 학생의 수를 1로 맞춘다
                    }
                }
            } else if(key == max) { //끝 번호의 학생이
                if(map.get(key) == 0) { //체육복이 없는 학생의 경우
                    if(map.get(key-1) == 2) { //이전번호 학생이 여벌이 있으면
                        map.put(key, 1);
                        map.put(key-1, 1);
                    }
                }
            } else { //중간 번호의 학생이
                if(map.get(key) == 0) { //체육복이 없는 학생의 경우
                    if(map.get(key-1) == 2) { //이전번호 학생이 여벌이 있으면
                        map.put(key, 1);
                        map.put(key-1, 1);
                    }
                    else if(map.get(key+1) == 2) { //다음번호 학생이 여벌이 있으면
                        map.put(key, 1);
                        map.put(key+1, 1);
                    }
                }
            }
        }

        for(int key : map.keySet()) { //체육복이 1벌이상인 학생을 찾으면서 answer를 증가시킨다
            if(map.get(key) > 0){
                answer++;
            }
        }
        
        return answer;
    }
}
```
